// Cosinus-√Ñhnlichkeit zwischen zwei Vektoren
function cosineSimilarity(vec1, vec2) {
  let dotProduct = 0;
  let norm1 = 0;
  let norm2 = 0;
  
  for (let i = 0; i < vec1.length; i++) {
    dotProduct += vec1[i] * vec2[i];
    norm1 += vec1[i] * vec1[i];
    norm2 += vec2[i] * vec2[i];
  }
  
  return dotProduct / (Math.sqrt(norm1) * Math.sqrt(norm2));
}

// Ergebnis: 0.0 (orthogonal) bis 1.0 (identisch)
üí° Mathematischer Hintergrund: Die Cosinus-√Ñhnlichkeit misst den Winkel zwischen zwei Vektoren im hochdimensionalen Raum. Je kleiner der Winkel, desto √§hnlicher sind die Inhalte.

// Complete Graph Generation Pipeline
async generateSemanticGraph(tenantId) {
  // 1. Lade alle Prozesse mit Embeddings
  const processes = await Process.find({
    tenantId,
    embedding: { $exists: true }
  }).lean();
  
  // 2. Berechne √Ñhnlichkeitsmatrix
  const nodes = [];
  const links = [];
  
  for (let i = 0; i < processes.length; i++) {
    nodes.push({
      id: processes[i]._id,
      title: processes[i].title,
      type: 'process',
      tags: processes[i].tagWeights
    });
    
    for (let j = i + 1; j < processes.length; j++) {
      const similarity = calculateHybridSimilarity(
        processes[i], 
        processes[j]
      );
      
      if (similarity > 0) {
        links.push({
          source: processes[i]._id,
          target: processes[j]._id,
          weight: similarity,
          type: 'semantic'
        });
      }
    }
  }
  
  // 3. Cache results
  await redis.setex(
    `graph:${tenantId}`, 
    3600, 
    JSON.stringify({ nodes, links })
  );
  
  return { nodes, links };
}
